Definition zexp (x : R) x_ (n : Z) : R :=
  match n with
  | Z0 => 1:R
  | Zpos p => x[^]nat_of_P p
  | Zneg p => (1[/] x[//]x_) [^]nat_of_P p
  end.

End Zexp_def.

Notation "( x ‡ Hx ) ^ n" := (zexp x Hx n) (at level 0).
