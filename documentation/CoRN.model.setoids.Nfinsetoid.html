Setoids of the first n natural numbers


Record F (n:nat):Set:=
{F_crr:> N ;
F_prf: F_crr<n
}.

Definition Feq (n : N) : F n -> F n -> Prop.
Defined.

Definition Fap (n : N) : F n -> F n -> CProp.
Defined.

Lemma Fap_irreflexive : forall n : N, irreflexive (Fap n).

Lemma Fap_symmetric : forall n : N, Csymmetric (Fap n).

Lemma Fap_cotransitive : forall n : N, cotransitive (Fap n).

Lemma Fap_tight : forall n : N, tight_apart (Feq n) (Fap n).

Definition less (n : N) :=
  Build_is_CSetoid (F n) (Feq n) (Fap n) (Fap_irreflexive n)
    (Fap_symmetric n) (Fap_cotransitive n) (Fap_tight n).

Definition CSetoid_of_less (n : N) : CSetoid :=
  Build_CSetoid (F n) (Feq n) (Fap n) (less n).
