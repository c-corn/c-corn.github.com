Section SeqProperties.

Definition seq_pos (x : N -> IR) := forall n : N, 0 [<] x n.
Definition seq_inf_sum (x : N -> IR) :=
forall M : IR, {N : N | forall m : N, N <= m -> M [<] ∑n x m}.

Lemma One_part_sum : forall (m : N), ∑n (fun n : N => 1) m [=] nring m.

Lemma One_seq_is_pos : seq_pos (fun n : N => 1).


Lemma One_seq_is_inf_sum : seq_inf_sum (fun n : N => 1).


Lemma seq_pos_imp_sum_pos : forall (x : N -> IR), seq_pos x -> forall n, 0 [<] ∑n x (S n).


Lemma seq_pos_imp_sum_pos' :
    forall (x : N -> IR) (H1 : seq_pos x) (n m : N) (H2 : m < n),
    0 [<] ∑ m n x.


Lemma seq_pos_imp_ap_zero : forall (x : N -> IR), seq_pos x -> forall n, ∑n x (S n) [#] 0.

Lemma seq_inf_sum_imp_div_small :
forall (x : N -> IR) (H1 : seq_inf_sum x) (H2: seq_pos x) (C e : IR)
(H4 : 0 [<] e), { N : N |
  forall m : N, N <= m -> AbsSmall e (C [/](∑n x (S m)) [//] (seq_pos_imp_ap_zero x H2 m))}.


Lemma seq_inf_sum_ratio_bound :
forall (y : N->IR) (H2 : seq_pos y) (m N1: N) (H3: S N1 < m),
AbsSmall 1 (∑ (G:=IR) (S N1) m (fun k : N => y k)[/]
                  ∑n y (S m)[//]seq_pos_imp_ap_zero y H2 m).


End SeqProperties.
